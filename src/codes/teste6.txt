let
    ! Declaração de variáveis
    var x : Integer;
    var y : Integer;
    var flag : Boolean;
in
    begin
    
        ! Inicialização
        x := 10;
        y := 5;
    
        ! Operações aritméticas
        x := x + y * 2;
    
        ! Condicional
        if x > y then
            flag := true
        else
            flag := false
    
        ! Loop while
        while x > 0 do
        begin
            ! decremento
            x := x - 1
        end
    
    end
